   #pragma once
   
   namespace gpa {
   namespace serialization {
   
   // Obtaining array length
   template<typename T>
   inline size_t GetNullTerminatedSize(T* array)
   {
       if (array == nullptr) {
           return 0;
       }
       size_t ii = 0;
       while (array[ii++] != T(0)) {
       }
   
       return ii;
   }
   
   template<typename T>
   inline size_t GetSizeOfArrayWithEndElem(T* array, T const endElem)
   {
       if (array == nullptr) {
           return 0;
       }
       size_t ii = 0;
       while (memcmp(&array[ii++], &endElem, sizeof(T)) != 0) {
       }
   
       return ii;
   }
   
   }  // namespace serialization
   }  // namespace gpa
