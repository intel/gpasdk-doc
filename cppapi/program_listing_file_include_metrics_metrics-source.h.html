#pragma once

#include <cstdint>

#include "igpa-metrics-config.h"

#include "metrics/types.h"

namespace gpa {
namespace metrics {

class IMetricGroup;

class GPA_METRICS_CLASS_EXPORT IMetricsSource
{
public:
    virtual ~IMetricsSource();

    virtual bool IsAvailable() const = 0;

    virtual uint32_t NumAvailableMetrics() const = 0;

    virtual MetricDesc const& MetricDescription(uint32_t const metricIndex) const = 0;

    virtual uint32_t NumMetricGroups() const = 0;
    virtual uint32_t NumMetricsInGroup(uint32_t const metricGroup) const = 0;

    virtual uint32_t GroupRawSampledReportSize(uint32_t const metricGroup) const = 0;

    virtual int32_t GroupContainingMetric(uint32_t const metricId) const = 0;

    virtual IMetricGroup const* MetricGroup(uint32_t metricGroupIndex) const = 0;
};

}  // namespace metrics
}  // namespace gpa
